//交换两变量的值的练习
#include<stdio.h>
#include<math.h>
void swap(int* pa,int* pb)
{
	int tem = 0;
	tem = *pa;
	*pa = *pb;
	*pb = tem;
}

int main()
{
	int a = 10;
	int b = 20;
	swap(&a, &b);
	printf("a = %d,b = %d",a,b);
	return 0;
}
